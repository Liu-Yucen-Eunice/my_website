# Use read_excel to read it as dataframe
bike0 <- read_excel(bike.temp,
                   sheet = "Data",
                   range = cell_cols("A:B"))

# change dates to get year, month, and week
bike <- bike0 %>% 
  clean_names() %>% 
  rename (bikes_hired = number_of_bicycle_hires) %>% 
  mutate (year = year(day),
          month = lubridate::month(day, label = TRUE),
          week = isoweek(day))
# Creating monthly data from 2016-19 and finding mean of biked hired
monthly_avg_1619 <- bike %>% 
  filter(year == c(2016:2019)) %>% 
  group_by(month) %>% 
  summarise(mean_1619 = mean(bikes_hired))

# Creating yearly data from 2016-21 and finding mean of biked hired
monthly_avg_by_year <- bike %>%
  filter(year %in% c(2016:2021)) %>%
  group_by(month,year) %>%
  summarise(mean_bikes = mean(bikes_hired)) %>% 
  arrange(year,month)

# Conflating the above 2 datasets
bike_monthly_plot <- merge(x=monthly_avg_1619,y=monthly_avg_by_year, by = "month",all.x = TRUE) %>% 
  mutate(increase = ifelse(mean_bikes > mean_1619, mean_bikes-mean_1619,0),
         decrease = ifelse(mean_bikes < mean_1619, mean_bikes-mean_1619,0))

# Creating plot
bike_monthly_plot %>% 
  ggplot(aes(x = month)) +
  geom_ribbon(aes(ymin = mean_1619,
                  ymax = mean_1619 + increase),
                  fill = 'darkgreen',
              group=1, alpha = 0.15) +
  geom_ribbon(aes(ymin = mean_1619 + decrease,ymax=mean_1619),fill = 'red',group=1, alpha = 0.25) +
  geom_line(aes(y = mean_1619, group=1),color = 'blue',size = 1) +
  geom_line(aes(y = mean_bikes, group=1), color = 'grey30') +
  facet_wrap(~year) +
  labs(title = 'Monthly Changes in Tfl Bike Rentals', subtitle = 'Change from monthly average shown in blue and calculated between 2016 -2019', caption= "Source: TfL, London Data Store", y = 'Bike Rentals', x = 'Month') +
  theme(axis.text.x = element_text(angle = 90, vjust = 0.5, hjust=1)) +
  NULL

The second one looks at percentage changes from the expected level of weekly rentals. The two grey shaded rectangles correspond to Q2 (weeks 14-26) and Q4 (weeks 40-52).

# Compute the mean of number of bikes hired for each week during the time period 2016-19
weekly_mean_1619 <- bike %>% 
  filter(year %in% c(2016,2017,2018,2019)) %>% 
  group_by(week) %>% 
  summarise(w_mean_1619 = mean(bikes_hired))

# Compute mean of number of bikes hired for each week in each year during the time period 2016-21
weekly_mean_yearly <- bike %>% 
  filter(year %in% c(2016,2017,2018,2019,2020,2021), week %in% c(1:52))%>% 
  group_by(year,week) %>% 
  summarise(w_mean_yearly = mean(bikes_hired))

week53<-bike[bike$week == "53" & bike$year %in% c("2020","2021"),] %>% 
  mutate(year=2020) %>% 
  group_by(year, week) %>% 
  summarise(w_mean_yearly = mean(bikes_hired))
weekly_mean_yearly[nrow(weekly_mean_yearly)+1,] = week53

# Join the two datasets to create dataset used in plotting, and add some features needed
bike_weekly_plot <- merge(x = weekly_mean_yearly, y = weekly_mean_1619,by = 'week') %>% 
  mutate(percent_change = (w_mean_yearly - w_mean_1619)/w_mean_1619*100,
         percent_increase = ifelse(percent_change>0,percent_change,0),
         percent_decrease = ifelse(percent_change<0,percent_change,0),
         )

bike_weekly_plot %>% 
  ggplot(aes(x = week, y = percent_change)) +
  geom_line(aes(y = percent_increase), color = "black", group = 1, size = 0.8, alpha = 0.3) + 
  geom_line(aes(y = percent_decrease), color = "black", group = 1, size = 0.8, alpha = 0.3) + 
  geom_rect(aes(xmin = 10,xmax = 30, ymin = -75, ymax = 125), fill = 'grey95', alpha = 0.3) +
  geom_rect(aes(xmin = 40,xmax = 53, ymin = -75, ymax = 125), fill = 'grey95', alpha = 0.3) +
  geom_area(aes(y = percent_increase),fill='darkgreen',alpha = 0.3) +
  geom_area(aes(y = percent_decrease),fill='darkred', alpha=0.3) +
  geom_rug(sides = 'b',color = ifelse(bike_weekly_plot$percent_change>0,'darkgreen','red')) +
  facet_wrap(~year) +
  scale_x_continuous(breaks = c(13,26,39,53))

  labs(title = 'Weekly Changes in Tfl Bike Rentals', subtitle = '% change from weekly averages calculated between 2016-2019', x = 'Week',y = 'Percentage Change (%)') +
  theme_minimal() +
    NULL
